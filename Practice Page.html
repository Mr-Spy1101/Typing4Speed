#include
  using namespace std;

  #define m 998244353
  #define add(x,y) (x + y) % m
  #define multiple(x, y) (x*y) % m

  unsigned long long shift(unsigned long long x) 
  {
    unsigned long long y = x%10;
    x /= 10;
    unsigned long long shift = 100;
    while (x > 0)
    {
      y += (x % 10) * shift;
      shift *= 100;
      x /= 10;
    }
    return y % m;
  }

  int main()
  {
    int n; cin >> n;
    unsigned long long temp;
    unsigned long long cal;
    unsigned long long shifted;
    unsigned long long result = 0;

    for (int i = 0; i < n; i++) 
    {
      cin >> temp;
      shifted = shift(temp);
      cal = add(multiple(shifted,n),multiple(shifted * 10,n));
      result = add(result, cal);
    }

    cout << result % m << endl;
    return 0;
  }
