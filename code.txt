void BubbleSort(int ArrayName[], const int ArraySize){
    int Temp;
    bool IsFound = false;
    for (int i = 1; i < ArraySize; i++)
	{
        for (int j = 0; j < ArraySize - i; j++)
		{
            if (ArrayName[j] > ArrayName[j + 1])
			{
                Temp = ArrayName[j];
                ArrayName[j] = ArrayName[j + 1];
                ArrayName[j + 1] = Temp;
                IsFound = true;
            }
        }
        if (!IsFound)
            break;
    }
}
